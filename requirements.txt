import math
from typing import List, Optional, Dict, Any

from hydraulik.widerstand import berechne_hydraulischen_widerstand
# Gemeinsame Membran-Helpers aus dem Parallelsimulator importieren
from app import (
    _berechne_tcf,
    _berechne_a_wert,
    _berechne_permeat_hydraulik,
    _iteriere_membranmodul,
    OSMOTISCHER_DRUCKKOEFF,
    SPACER_DRUCKVERLUST_FAKTOR,
)

# --- Konstanten ---
CD_BLENDE = 0.6                 # Durchflusskoeffizient Kreisblende [-]
DICHTE_WASSER_KG_M3 = 1000.0   # kg/m³
SOLVER_ITERATIONEN = 60         # Bisektionsschritte für numerische Nullpunkt-Suche
FEED_MIN_LH = 5.0               # Untere Suchgrenze Feedfluss [l/h]
MAX_RECOVERY_FAKTOR = 0.99      # Sicherheitsabstand zur Pumpengrenze


def berechne_drossel_druckabfall(flow_lh: float, drossel_mm: float) -> float:
    """
    Berechnet den Druckabfall [bar] über eine Kreisblende.

    Basiert auf der realen Blenden-Gleichung mit Cd = 0,6.

    Args:
        flow_lh:     Volumenstrom [l/h]
        drossel_mm:  Drosseldurchmesser [mm]

    Returns:
        Druckabfall [bar]; 9999 bei ungültigen Eingaben
    """
    if flow_lh <= 0.001 or drossel_mm <= 0:
        return 9999.0

    q_ms = (flow_lh / 1000.0) / 3600.0
    d_m = drossel_mm / 1000.0
    area_m2 = math.pi * (d_m / 2.0) ** 2

    v_spalt = q_ms / (area_m2 * CD_BLENDE)
    delta_p_pa = (DICHTE_WASSER_KG_M3 * v_spalt ** 2) / 2.0
    return delta_p_pa / 1e5


def berechne_pumpendruck(flow_lh: float, p_max: float, q_max: float) -> float:
    """
    Berechnet den Pumpenförderdruck [bar] über eine parabolische Kennlinie.

    Modell: P = P_max · (1 - (Q / Q_max)²)  (Kreiselpumpen-Näherung)

    Args:
        flow_lh: Aktueller Volumenstrom [l/h]
        p_max:   Nullförderdruck der Pumpe [bar]
        q_max:   Freiförderstrom der Pumpe [l/h]

    Returns:
        Förderdruck [bar]; 0 wenn Strom ≥ Freiförderstrom
    """
    if flow_lh >= q_max:
        return 0.0
    return p_max * (1.0 - (flow_lh / q_max) ** 2)


def _berechne_konzentrat_druck(
    membran_daten: List[Dict],
    leitungen_konz: List[Dict],
    leitung_out: Optional[Dict],
    end_konzentrat_flow: float,
    p_effektiv_start: float,
    anzahl_membranen: int,
) -> float:
    """Berechnet den Druck vor dem Regelventil aus dem Konzentrat-Pfad."""
    p_nach_spacer = p_effektiv_start - SPACER_DRUCKVERLUST_FAKTOR

    if anzahl_membranen == 1:
        if leitung_out:
            r_out = berechne_hydraulischen_widerstand(
                leitung_out["d"], leitung_out["l"], [], leitung_out["b"]
            )
            return p_nach_spacer - r_out * ((end_konzentrat_flow / 1000.0) / 3600.0) ** 2 / 1e5
        return p_nach_spacer

    # Mehrere Membranen: max. Druckverlust im Zweig ist maßgebend
    p_verluste_zweige = [
        berechne_hydraulischen_widerstand(cfg["d"], cfg["l"], [], cfg["b"])
        * ((membran_daten[i]["_q_c"] / 1000.0) / 3600.0) ** 2 / 1e5
        for i, cfg in enumerate(leitungen_konz)
    ]
    p_t_stueck = p_nach_spacer - (max(p_verluste_zweige) if p_verluste_zweige else 0.0)

    if leitung_out:
        r_out = berechne_hydraulischen_widerstand(
            leitung_out["d"], leitung_out["l"], [], leitung_out["b"]
        )
        return p_t_stueck - r_out * ((end_konzentrat_flow / 1000.0) / 3600.0) ** 2 / 1e5
    return p_t_stueck


def simuliere_parallel_drossel(
    flow_fractions: List[float],
    membran_namen: List[str],
    drossel_vorgabe_mm: float,
    m_flaeche: float,
    m_test_flow: float,
    m_test_druck: float,
    m_rueckhalt: float,
    tds_feed: float,
    temp: float,
    p_max: float,
    q_max: float,
    p_zulauf: float,
    r_saug: float,
    r_druck_haupt: float,
    r_netzwerk: float,
    hat_t_stueck: bool,
    leitungen_konz: List[Dict],
    leitung_out: Optional[Dict],
    p_leitungen_konz: List[Dict],
    p_leitung_out: Optional[Dict],
    p_schlauch_out: Optional[Dict],
) -> Dict[str, Any]:
    """
    Simuliert ein RO-Parallelsystem mit Pumpen-Kennlinie und Drossel-Regelung.

    Verwendet einen numerischen Bisektionssolver, um den Betriebspunkt zu finden,
    an dem der Druck hinter der Drossel auf 0 bar sinkt (physikalisches Gleichgewicht).

    Args:
        drossel_vorgabe_mm: Fest vorgegebener Drosseldurchmesser [mm]
        p_max:              Nullförderdruck der Pumpe [bar]
        q_max:              Freiförderstrom der Pumpe [l/h]
        p_zulauf:           Zulaufdruck vor der Pumpe [bar]
        (weitere Parameter: siehe simuliere_parallel)

    Returns:
        Dict mit Simulationsergebnissen oder {"error": <Meldung>} bei Fehler.
    """
    anzahl_membranen = len(flow_fractions)
    if anzahl_membranen == 0:
        return {"error": "Keine Membranen gefunden."}

    tcf_real = _berechne_tcf(temp)
    a_wert = _berechne_a_wert(m_test_flow, m_flaeche, m_test_druck)

    r_p_out, r_p_schlauch, p_back_height, r_p_branches = _berechne_permeat_hydraulik(
        p_leitungen_konz, p_leitung_out, p_schlauch_out, anzahl_membranen
    )

    # --- Bisektions-Solver: Suche Feedfluss, bei dem Restdruck hinter Drossel = 0 ---
    feed_min = FEED_MIN_LH
    feed_max = min(20000.0, q_max * MAX_RECOVERY_FAKTOR)
    best_result: Dict[str, Any] = {"error": "Solver konnte kein Gleichgewicht finden."}

    for _ in range(SOLVER_ITERATIONEN):
        q_feed_start_lh = (feed_min + feed_max) / 2.0
        q_ms = (q_feed_start_lh / 1000.0) / 3600.0

        # Druckaufbau: Zulauf → Saugverlust → Pumpe → Leitungsverluste
        p_verlust_saug = (r_saug * q_ms ** 2) / 1e5
        p_vor_pumpe = max(0.0, p_zulauf - p_verlust_saug)

        p_pumpen_boost = berechne_pumpendruck(q_feed_start_lh, p_max, q_max)
        p_aktuell = p_vor_pumpe + p_pumpen_boost

        p_verlust_druck_haupt = (r_druck_haupt * q_ms ** 2) / 1e5
        p_verlust_netzwerk = (r_netzwerk * q_ms ** 2) / 1e5 if hat_t_stueck else 0.0
        p_effektiv_start = p_aktuell - p_verlust_druck_haupt - p_verlust_netzwerk

        if p_effektiv_start <= 0.5:
            feed_max = q_feed_start_lh
            continue

        # Haupt-Gegendruck Permeat schätzen
        q_ms_p = ((q_feed_start_lh * 0.4) / 1000.0) / 3600.0
        p_back_main = (r_p_out + r_p_schlauch) * q_ms_p ** 2 / 1e5 + p_back_height

        # Membranmodule berechnen (gemeinsamer Helper)
        membran_daten = [
            _iteriere_membranmodul(
                name=membran_namen[i],
                f_in=q_feed_start_lh * flow_fractions[i],
                tds_feed=tds_feed,
                m_rueckhalt=m_rueckhalt,
                m_flaeche=m_flaeche,
                a_wert=a_wert,
                tcf_real=tcf_real,
                p_effektiv_start=p_effektiv_start,
                p_back_main=p_back_main,
                r_p_branch=r_p_branches[i],
            )
            for i in range(anzahl_membranen)
        ]

        total_permeat = sum(m["_q_p"] for m in membran_daten)
        total_permeat_salzfracht = sum(m["_q_p"] * m["_tds_p"] for m in membran_daten)
        end_konzentrat_flow = max(1e-6, q_feed_start_lh - total_permeat)

        # Konzentrat-Druck und Drossel-Abgleich
        p_vor_ventil = _berechne_konzentrat_druck(
            membran_daten, leitungen_konz, leitung_out,
            end_konzentrat_flow, p_effektiv_start, anzahl_membranen,
        )
        p_verlust_drossel = berechne_drossel_druckabfall(end_konzentrat_flow, drossel_vorgabe_mm)
        restdruck_nach_ventil = p_vor_ventil - p_verlust_drossel

        # Bisektion: positiver Restdruck → Feed erhöhen, negativer → Feed senken
        if restdruck_nach_ventil > 0.0:
            feed_min = q_feed_start_lh
        else:
            feed_max = q_feed_start_lh

        avg_permeat_tds = total_permeat_salzfracht / total_permeat if total_permeat > 0 else 0.0
        final_konzentrat_tds = (
            (q_feed_start_lh * tds_feed - total_permeat_salzfracht) / end_konzentrat_flow
            if end_konzentrat_flow > 0 else tds_feed
        )

        # Interne Hilfsfelder bereinigen
        for m in membran_daten:
            m.pop("_q_p", None)
            m.pop("_tds_p", None)
            m.pop("_q_c", None)

        best_result = {
            "error": None,
            "q_feed_start_lh": q_feed_start_lh,
            "total_permeat": total_permeat,
            "total_permeat_tds": avg_permeat_tds,
            "final_konzentrat_tds": final_konzentrat_tds,
            "end_konzentrat_flow": end_konzentrat_flow,
            "membran_daten": membran_daten,
            "p_verlust_saug": p_verlust_saug,
            "p_verlust_druck_haupt": p_verlust_druck_haupt,
            "p_verlust_netzwerk": p_verlust_netzwerk,
            "p_effektiv_start": p_effektiv_start,
            "konzentrat_druck_verlauf": max(0.0, p_vor_ventil),
            "abzubauender_druck": min(max(0.0, p_vor_ventil), p_verlust_drossel),
            "empfohlene_drossel_mm": drossel_vorgabe_mm,
            "realer_pumpendruck": p_aktuell,
        }

    return best_result
